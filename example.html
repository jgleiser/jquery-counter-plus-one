Example page for jQuery Counter Plus One plugin

Default, no options, following required markup

00053

      
  <!-- Default HTML markup -->
  <div class="jcpo-counter counter1">
    <p>
      00053
    </p>
  </div>
  
  <!-- Default trigger event to .jcpo-plusone on click -->
  <button type="button" class="jcpo-plusone">
    Counter + 1
  </button>
  
  <script>
    $(".counter1").counter();
  </script>

Value script defined

use same control as above!

      
  <!-- Container for value defined in script -->
  <div class="jcpo-counter counter2"></div>
  
  <script> 
    $(".counter2").counter({
        val: '06787'
    });
  </script>

Custom trigger

mouse over me! 
      
  $('.counter3').counter({
      val: '392',
      triggerEvent: 'mouseover',
      triggerElement: '.jcpo-plusone-custom1'
  });

Another custom trigger

this is time based!

      
  $('.counter4').counter({
      val: '2001',
      countTrigger: function ($counter) {
          setInterval(function() {
              $.counter.plusOne($counter);
          }, 1000);
      }
  });

  /*
    Note: countTrigger can be anything that triggers
    $.counter.plusOne($counter) but should always be wraped
    in function ($counter) {}
  */
